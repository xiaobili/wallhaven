      2. 实现 `useWallpaperList()` 函数：
         ```typescript
         export function useWallpaperList(): UseWallpaperListReturn {
           const store = useWallpaperStore()
           const { showError } = useAlert()
           
           // fetch: 调用 wallpaperService.search()，更新 store
           // loadMore: 计算下一页，调用 wallpaperService.search()，追加数据
           // reset: 重置 store 状态
           // saveCustomParams: 调用 wallpaperService.saveQueryParams()
           // loadSavedParams: 调用 wallpaperService.loadQueryParams()
           
           return {
             wallpapers: computed(() => store.totalPageData),
             loading: computed(() => store.loading),
             error: computed(() => store.error),
             queryParams: computed(() => store.queryParams),
             savedParams: computed(() => store.savedParams),
             fetch,
             loadMore,
             reset,
             saveCustomParams,
             loadSavedParams,
           }
         }
         ```

      3. fetch 方法实现：
         ```typescript
         const fetch = async (params: GetParams | null): Promise<boolean> => {
           store.loading = true
           store.error = false
           
           const result = await wallpaperService.search(params)
           
           if (!result.success) {
             showError(result.error?.message || '获取壁纸失败')
             store.error = true
             store.loading = false
             return false
           }
           
           store.queryParams = params
           store.totalPageData = {
             sections: [result.data],
             totalPage: result.data.meta.last_page,
             currentPage: result.data.meta.current_page,
           }
           store.loading = false
           return true
         }
         ```

      4. loadMore 方法实现：
         ```typescript
         const loadMore = async (): Promise<boolean> => {
           if (!store.queryParams || store.loading) return false
           
           // 检查是否已加载所有页面
           if (store.totalPageData.totalPage > 0 &&
               store.totalPageData.currentPage >= store.totalPageData.totalPage) {
             return false
           }
           
           store.loading = true
           
           const nextPage = store.totalPageData.currentPage + 1
           const params = { ...store.queryParams, page: nextPage }
           
           const result = await wallpaperService.search(params)
           
           if (!result.success) {
             showError(result.error?.message || '加载更多失败')
             store.loading = false
             return false
           }
           
           store.totalPageData = {
             ...store.totalPageData,
             sections: [...store.totalPageData.sections, result.data],
             currentPage: result.data.meta.current_page,
           }
           store.loading = false
           return true
         }
         ```

      5. 其他方法实现（reset, saveCustomParams, loadSavedParams）参考 RESEARCH.md
