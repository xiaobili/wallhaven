### Task 4.3: Add handleToggleFavorite function

<read_first>
- `src/views/OnlineWallpaper.vue` — Existing methods section
- `src/composables/favorites/useFavorites.ts` — add and remove function signatures
</read_first>

<action>
Add to `src/views/OnlineWallpaper.vue` methods:

```typescript
/**
 * Handle favorite button click - show dropdown
 */
const handleToggleFavorite = (item: WallpaperItem, event: MouseEvent): void => {
  dropdownWallpaper.value = item
  // Position dropdown near click
  const rect = (event.target as HTMLElement).getBoundingClientRect()
  dropdownPosition.value = {
    x: rect.left,
    y: rect.bottom + 4  // 4px below button
  }
  showFavoriteDropdown.value = true
}

/**
 * Close favorite dropdown
 */
const closeFavoriteDropdown = (): void => {
  showFavoriteDropdown.value = false
  dropdownWallpaper.value = null
}

/**
 * Handle add to collection from dropdown
 */
const handleAddToCollection = async (collectionId: string): Promise<void> => {
  if (!dropdownWallpaper.value) return
  const success = await addFavorite(dropdownWallpaper.value.id, collectionId, dropdownWallpaper.value)
  if (success) {
    showSuccess('已添加到收藏')
  }
}

/**
 * Handle remove from collection from dropdown
 */
const handleRemoveFromCollection = async (collectionId: string): Promise<void> => {
  if (!dropdownWallpaper.value) return
  const success = await removeFavorite(dropdownWallpaper.value.id, collectionId)
  if (success) {
    showSuccess('已从收藏移除')
  }
}

/**
 * Handle move between collections
 */
const handleMoveToCollection = async (fromCollectionId: string, toCollectionId: string): Promise<void> => {
  if (!dropdownWallpaper.value) return
  const success = await moveFavorite(dropdownWallpaper.value.id, fromCollectionId, toCollectionId)
  if (success) {
    showSuccess('已移动到其他收藏夹')
  }
}
</action>

- closeFavoriteDropdown function exists
