3. Add implementation after the existing `getDefaults` function:
```typescript
// === Editable Settings for Form Binding ===

/**
 * Local editable copy for form binding
 * Independent of store state - changes don't affect store until saveChanges()
 */
const editableSettings = ref<AppSettings>(getDefaults()) as Ref<AppSettings>

/**
 * Initialize/sync editableSettings from store
 * Call this when component mounts or after reset
 */
const startEdit = (): void => {
  Object.assign(editableSettings.value, store.settings)
}

/**
 * Discard local changes, sync from store
 */
const discardChanges = (): void => {
  Object.assign(editableSettings.value, store.settings)
}

/**
 * Save local changes to store and persist
 * @returns Whether save was successful
 */
const saveChanges = async (): Promise<boolean> => {
  // Validate before saving
  const result = await update(editableSettings.value)
  if (result) {
    // Sync local copy with store after successful save
    Object.assign(editableSettings.value, store.settings)
  }
  return result
}

/**
 * Check if there are unsaved local changes
 */
const isDirty = computed(() => {
  return JSON.stringify(editableSettings.value) !== JSON.stringify(store.settings)
})
```

4. Update the return statement to include new fields:
