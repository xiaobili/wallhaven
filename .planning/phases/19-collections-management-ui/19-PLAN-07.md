<template>
  <div class="favorites-page">
    <!-- Sidebar with collection list -->
    <CollectionSidebar
      v-model:selected-id="selectedCollectionId"
      @select="handleSelectCollection"
    />
    
    <!-- Main content area -->
    <div class="favorites-content">
      <!-- Header -->
      <div class="content-header">
        <h2>{{ currentCollectionName }}</h2>
        <p
          v-if="currentCollection"
          class="collection-info"
        >
          {{ currentCount }} 张壁纸
        </p>
      </div>
      
      <!-- Empty state for selected collection -->
      <div
        v-if="!loading && currentCount === 0"
        class="empty-content"
      >
        <i class="fas fa-images" />
        <h3>暂无收藏</h3>
        <p>在在线壁纸页面点击收藏图标，将壁纸添加到此收藏夹</p>
      </div>
      
      <!-- Placeholder for Phase 21 content (wallpaper grid) -->
      <div
        v-else
        class="content-grid"
      >
        <!-- Phase 21 will add wallpaper grid here -->
        <p class="placeholder-text">
          收藏内容将在 Phase 21 实现
        </p>
      </div>
    </div>
    
    <!-- Alert component -->
    <Alert
      v-if="alert.visible"
      :type="alert.type"
      :message="alert.message"
      :duration="alert.duration"
      @close="hideAlert"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import CollectionSidebar from '@/components/favorites/CollectionSidebar.vue'
import Alert from '@/components/Alert.vue'
import { useCollections, useFavorites, useAlert } from '@/composables'
import type { Collection } from '@/types'

// Composables
const { collections, loading, load: loadCollections, getById, getDefault } = useCollections()
const { favorites, load: loadFavorites, getByCollection } = useFavorites()
const { alert, hideAlert } = useAlert()

// State
const selectedCollectionId = ref<string>('')

// Current collection
const currentCollection = computed(() => {
  if (!selectedCollectionId.value) return getDefault()
  return getById(selectedCollectionId.value)
})

const currentCollectionName = computed(() => 
  currentCollection.value?.name || '我的收藏'
)

const currentCount = computed(() => {
  if (!currentCollection.value) return 0
  return getByCollection(currentCollection.value.id).length
})

// Handlers
const handleSelectCollection = (collection: Collection) => {
  selectedCollectionId.value = collection.id
}

// Initialize
onMounted(async () => {
  await loadCollections()
  await loadFavorites()
  
  // Select default collection if none selected
  if (!selectedCollectionId.value) {
    const defaultCollection = getDefault()
    if (defaultCollection) {
      selectedCollectionId.value = defaultCollection.id
    }
  }
})
</script>
