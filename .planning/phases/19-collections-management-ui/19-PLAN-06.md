<template>
  <div class="collection-sidebar">
    <div class="sidebar-header">
      <h3>收藏夹</h3>
      <button
        class="create-btn"
        title="新建收藏夹"
        @click="showCreateModal = true"
      >
        <i class="fas fa-plus" />
      </button>
    </div>
    
    <div class="sidebar-content">
      <div
        v-if="loading"
        class="loading-state"
      >
        <i class="fas fa-spinner fa-spin" />
        <span>加载中...</span>
      </div>
      
      <div
        v-else-if="collections.length === 0"
        class="empty-state"
      >
        <i class="fas fa-folder-open" />
        <p>还没有收藏夹</p>
        <p class="hint">
          点击上方 + 按钮创建
        </p>
      </div>
      
      <div
        v-else
        class="collection-list"
      >
        <CollectionItem
          v-for="collection in collections"
          :key="collection.id"
          :collection="collection"
          :count="getCollectionCount(collection.id)"
          :is-active="selectedId === collection.id"
          @select="handleSelect"
          @rename="handleRename"
          @delete="handleDelete"
        />
      </div>
    </div>
    
    <!-- Create Modal -->
    <CreateCollectionModal
      :visible="showCreateModal"
      :existing-names="existingNames"
      @close="showCreateModal = false"
      @confirm="handleCreateConfirm"
    />
    
    <!-- Rename Modal -->
    <RenameCollectionModal
      :visible="showRenameModal"
      :current-name="renameTarget?.name || ''"
      :collection-id="renameTarget?.id || ''"
      :existing-names="existingNames"
      @close="handleRenameClose"
      @confirm="handleRenameConfirm"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import CollectionItem from './CollectionItem.vue'
import CreateCollectionModal from './CreateCollectionModal.vue'
import RenameCollectionModal from './RenameCollectionModal.vue'
import { useCollections, useFavorites } from '@/composables'
import type { Collection } from '@/types'

interface Props {
  selectedId?: string
}

const props = withDefaults(defineProps<Props>(), {
  selectedId: ''
})

const emit = defineEmits<{
  'update:selected-id': [id: string]
  select: [collection: Collection]
}>()

// Composables
const { collections, loading, load, create, rename, delete: deleteCollection } = useCollections()
const { favorites, load: loadFavorites } = useFavorites()

// Modal states
const showCreateModal = ref(false)
const showRenameModal = ref(false)
const renameTarget = ref<Collection | null>(null)

// Existing names for validation
const existingNames = computed(() => 
  collections.value.map(c => c.name)
)

// Get favorite count for a collection
const getCollectionCount = (collectionId: string): number => {
  return favorites.value.filter(f => f.collectionId === collectionId).length
}

// Handlers
const handleSelect = (collection: Collection) => {
  emit('update:selected-id', collection.id)
  emit('select', collection)
}

const handleRename = (collection: Collection) => {
  renameTarget.value = collection
  showRenameModal.value = true
}

const handleDelete = async (collection: Collection) => {
  const count = getCollectionCount(collection.id)
  const message = count > 0
    ? `确定要删除收藏夹 "${collection.name}" 吗？\n该收藏夹内有 ${count} 张壁纸，将一并移除。`
    : `确定要删除收藏夹 "${collection.name}" 吗？`
  
  const confirmed = window.confirm(message)
  if (!confirmed) return
  
  const success = await deleteCollection(collection.id)
  if (success) {
    // Refresh favorites list
    await loadFavorites()
  }
}

const handleCreateConfirm = async (name: string) => {
  showCreateModal.value = false
  await create(name)
}

const handleRenameClose = () => {
  showRenameModal.value = false
  renameTarget.value = null
}

const handleRenameConfirm = async (id: string, name: string) => {
  showRenameModal.value = false
  renameTarget.value = null
  await rename(id, name)
}

// Load data on mount
onMounted(async () => {
  await load()
  await loadFavorites()
})
</script>
