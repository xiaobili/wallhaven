// Expected structure
import type { WallpaperItem } from '@/types'

class FavoritesServiceImpl {
  private cachedFavorites: FavoriteItem[] | null = null
  private cachedData: FavoritesData | null = null

  async getAll(): Promise<IpcResponse<FavoriteItem[]>>
  async getByCollection(collectionId: string): Promise<IpcResponse<FavoriteItem[]>>
  async isFavorite(wallpaperId: string): Promise<IpcResponse<boolean>>
  async getCollectionsForWallpaper(wallpaperId: string): Promise<IpcResponse<Collection[]>>
  async add(wallpaperId: string, collectionId: string, wallpaperData: WallpaperItem): Promise<IpcResponse<FavoriteItem>>
  async remove(wallpaperId: string, collectionId: string): Promise<IpcResponse<void>>
  async move(wallpaperId: string, fromCollectionId: string, toCollectionId: string): Promise<IpcResponse<FavoriteItem>>
  clearCache(): void
}

export const favoritesService = new FavoritesServiceImpl()

## Notes

- Both collectionsService and favoritesService are independent and both depend only on favoritesRepository
- No cross-dependency between services (as decided in D-07)
- Cache invalidation on write operations ensures data consistency
- The add() method creates FavoriteItem internally with wallpaperId, collectionId, addedAt, and wallpaperData
